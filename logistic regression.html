Sigmoid function¶

In [7]:
def sigmoid(x):
    s = 1 / (1 + np.exp(-x))
    return s
In [8]:
# Calculate the objective function value
# Inputs:
#     w: d-by-1 matrix
#     x: n-by-d matrix
#     t: n-by-1 matrix
# Return:
#     objective function value (scalar)

def loglikelihood(w,x,t):
    y = np.dot(x,w)
    y = sigmoid(y)
    lny = np.log(y)
    log_likelihood= np.multiply(t,lny)+np.multiply((1-t),(np.log(1-y)))
    log_likelihood = np.sum(log_likelihood)
    return log_likelihood

# Calculate the gradient
# Inputs:
#     w: d-by-1 matrix
#     x: n-by-d matrix
#     y: n-by-1 matrix
# Return:
#     g: g: d-by-1 matrix, full gradient
def gradient(w, x, t):
    n, d = x.shape
    y = sigmoid(np.dot(x,w))
    t_y = t-y
    vec1 = np.multiply(t_y, x)
    g = np.sum(vec1, axis = 0).reshape(d,1)
    return g

# Gradient descent 
# Inputs:
#     x: n-by-d matrix
#     y: n-by-1 matrix
#     max_iter: integer, the maximal iterations
#     w: d-by-1 matrix, initialization of w
# Return:
#     w: d-by-1 matrix, the solution
#     loglikelihoods: a record of each iteration's loglikelihood value
def grad_descent(x, y, stepsize, max_iter=100, w=None):
    n, d = x.shape
    log_likelihoods = np.zeros(max_iter) # store the objective values
    if w is None:
#         w = np.zeros((d, 1)) # zero initialization
#         w = np.ones((d, 1)) # zero initialization
        w = np.random.normal(0,0.1,(d, 1)) # zero initialization
    
    for t in range(max_iter):
        log_likelihood = loglikelihood(w, x, y)
        log_likelihoods[t] = log_likelihood
#         print('Objective value at t=' + str(t) + ' is ' + str(objval))
        g = gradient(w, x, y)
#         print(g)
        w += stepsize * g

        print('Log likelihood value at t=' + str(t) + ' is ' + str(log_likelihood), ',train_acc=' + str(acc))
    
    return w, log_likelihoods, acc

def pred(x,w,t):
    n, d = x.shape
    y = sigmoid(np.dot(x,w))
    y[y>0.5]=1
    y[y<=0.5]=0
    correct = np.sum(t == y) 
    acc = 100.0 * float(correct) / n
    return acc
